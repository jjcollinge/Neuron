9   /**
10   * Encapsulates the session business logic
11   * @author JC
12   *
13   */
14  public class SessionActivity extends Activity {
15      
16      public SessionActivity(String name) {
17          super(name);
18          addService("SessionHandler", new SessionHandler());
19          addService("ActivityListener", new ActivityListener());
20      }
21      
22      public void setup(Configuration config) {
23          getService("SessionHandler").setup(config);
24          ActivityListener al = (ActivityListener) getService("ActivityListener");
25          al.addRequestListener((RequestListener) getService("SessionHandler"));
26          RegistrationRequestHandler requestHandler = (RegistrationRequestHandler) getService("RequestHandler");
27          if(requestHandler != null) {
28              requestHandler.addRegistrationListener((RegistrationListener) getService("SessionHandler"));
29          } else {
30              System.out.println("ERROR: cannot start service without registration handler");
31              stop();
32          }
33      }
