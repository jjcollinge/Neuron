47      @Test
48      public void testHandleRegistrationRequest() {
49          
50          // Given
51          Request request = new Request();
52          request.setData("{\"regAddress\":\"2077\",\"device\":{" +
53                          "\"manufacturer\": \"RaspberryPi\", " +
54                          "\"model\": \"B+\"," +
55                          "\"geo\": {"+
56                              "\"latitude\": 53.37831623,"+
57                              "\"longitude\": -1.4618752"+
58                          "},"+
59                          "\"sensors\": [],"+
60                          "\"actuators\": []"+
61                          "}}");
62          request.setFormat("json");
63          request.setProtocol("mqtt");
64          request.addHeader("topic", "1773");
65          
66          Device expectedDevice = new Device();
67          expectedDevice.setManufacurer("RaspberryPi");
68          expectedDevice.setModel("B+");
69          expectedDevice.setGeo(new GeoPoint(53.37831623, -1.4618752));
70          Registration expectedRegistration = new Registration();
71          expectedRegistration.setDevice(expectedDevice);
72          expectedRegistration.setRegistrationAddress("2077");
73          
74          RegistrationResponderMock responder = new RegistrationResponderMock();
75          
76          // When
77          handler.addRegistrationListener(responder);
78          handler.onRequest(request);
79  
80          // Should
81          assertEquals(expectedRegistration.getRegistrationAddress(), responder.getLastRegistration().getRegistrationAddress());
82          assertEquals(expectedRegistration.getDevice().getManufacturer(), responder.getLastRegistration().getDevice().getManufacturer());
83          assertEquals(expectedRegistration.getDevice().getModel(), responder.getLastRegistration().getDevice().getModel());
84      }
