3   import java.io.IOException;
4   import java.util.logging.Level;
5   import java.util.logging.Logger;
6   
7   import org.codehaus.jackson.JsonGenerationException;
8   import org.codehaus.jackson.JsonParseException;
9   import org.codehaus.jackson.map.JsonMappingException;
10  import org.codehaus.jackson.map.ObjectMapper;
11  
12  import com.neuron.api.data_access.ObjectMapperStrategy;
13  
14  /**
15   * Responsible for mapping a serialized registration to a POJO
16   * and back again.
17   * @author JC
18   *
19   */
20  public class JsonRegistrationMapper implements ObjectMapperStrategy<Registration, String> {
21  
22      private static final Logger log = Logger.getLogger(JsonRegistrationMapper.class
23              .getName());
24      
25      public String serialize(Registration source) {
26          
27          ObjectMapper mapper = new ObjectMapper();
28          String result = null;
29          
30          try {
31              result = mapper.writeValueAsString(source);
32          } catch (JsonGenerationException e) {
33              log.log(Level.WARNING, "Failed to parse Json: generation exception");
34          } catch (JsonMappingException e) {
35              log.log(Level.WARNING, "Failed to parse Json: mapping exception");
36          } catch (IOException e) {
37              log.log(Level.WARNING, "Failed to parse Json: io exception");
38          }
39          return result;      
40      }
41  
42      public Registration deserialize(String source) {
43  
44          ObjectMapper mapper = new ObjectMapper();
45          Registration reg = null;
46          
47          if(source != null) {
48              try {
49                  reg = mapper.readValue(source, Registration.class);
50              } catch (JsonParseException e) {
51                  log.log(Level.WARNING, "Failed to parse JSON: parse exception", e);
52              } catch (JsonMappingException e) {
53                  log.log(Level.WARNING, "Failed to parse JSON: mapping exception", e);
54              } catch (IOException e) {
55                  log.log(Level.WARNING, "Failed to parse JSON: io exception", e);
56              } 
57          }
58          
59          return reg;
60      }
