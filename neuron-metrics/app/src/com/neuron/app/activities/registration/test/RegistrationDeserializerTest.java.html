41      @Test
42      public void testDeserializingRegistration() {
43          
44          // Given
45          Device expectedDevice = new Device();
46          expectedDevice.setManufacurer("RaspberryPi");
47          expectedDevice.setModel("B+");
48          expectedDevice.setGeo(new GeoPoint(53.37831623, -1.4618752));
49          String registration = "{\"regAddress\":\"2077\",\"device\":{" +
50                                  "\"manufacturer\": \"RaspberryPi\", " +
51                                  "\"model\": \"B+\"," +
52                                  "\"geo\": {"+
53                                      "\"latitude\": 53.37831623,"+
54                                      "\"longitude\": -1.4618752"+
55                                  "},"+
56                                  "\"sensors\": [],"+
57                                  "\"actuators\": []"+
58                              "}}";
59          // When
60          Registration reg = deserializer.deserialize(registration);
61          
62          // Should
63          assertEquals(reg.getDevice().getManufacturer(), expectedDevice.getManufacturer());
64          assertEquals(reg.getDevice().getModel(), expectedDevice.getModel());
65          assertArrayEquals(reg.getDevice().getSensors().toArray(), expectedDevice.getSensors().toArray());
66          assertArrayEquals(reg.getDevice().getActuators().toArray(), expectedDevice.getActuators().toArray());
67          assertEquals(reg.getRegistrationAddress(), "2077");
68          
69      }
