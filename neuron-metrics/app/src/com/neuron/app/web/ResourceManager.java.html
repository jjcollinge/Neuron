1   package com.neuron.app.web;
2   
3   import java.util.HashMap;
4   
5   /**
6    * In order to avoid creating multiple resources per
7    * request and thus invoking creating multiple
8    * streams to and from the devices, the resources must
9    * be managed. The ResourceManager is responsible for
10   * storing resources with their URI as a key. This
11   * allows clients to grab any existing resource based
12   * only on their URI.
13   * @author JC
14   *
15   */
16  public class ResourceManager {
17  
18      private HashMap<String, Object> resources;
19      private static ResourceManager instance;
20      
21      private ResourceManager() {
22          resources = new HashMap<String, Object>();
23      }
24      
25      public static ResourceManager getInstance() {
26          if(instance == null) {
27              instance = new ResourceManager();
28          }
29          return instance;
30      }
31      
32      /**
33       * Store a new resource keyed by its uri
34       * @param uri The desired URI
35       * @param resource The resource to store
36       */
37      public void addResource(String uri, Object resource) {
38          resources.put(uri, resource);
39      }
40      
41      /**
42       * Remove a resource
43       * @param resource resource The resource to remove
44       */
45      public void removeResource(Object resource) {
46          resources.remove(resource);
47      }
48      
49      /**
50       * Return a resource keyed by its uri
51       * @param uri The resource identifier to get
52       * @return Object The resource
53       */
54      public Object getResource(String uri) {
55          return resources.get(uri);
56      }
57      
58  }
