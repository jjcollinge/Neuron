7   import javax.ws.rs.container.ContainerRequestContext;

9   import javax.ws.rs.container.PreMatching;
10  import javax.ws.rs.core.Response;
11  import javax.ws.rs.ext.Provider;
12  
13  
14  /**
15   * Applies a filter to any incoming requests. The filter will
16   * accept any preflight HTTP messages sent from the browser
17   * to authenticate CORS(Cross Origin Resource Sharing) which
18   * can sometime be blocked in languages such as javascript.
19   * @author JC
20   *
21   */
22  @Provider
23  @PreMatching
24  public class CORSRequestFilter implements ContainerRequestFilter {
25  
26      private final static Logger log = Logger.getLogger( CORSRequestFilter.class.getName() );
27      
28      public void filter(ContainerRequestContext requestContext)
29              throws IOException {
30          
31          log.log(Level.INFO, "Applying request filter");
32          
33          if( requestContext.getRequest().getMethod().equals( "OPTIONS" )) {
34              
35              log.log(Level.INFO, "Browser pre-flight HTTP Method OPTIONS detected!");
36              
37              requestContext.abortWith( Response.status( Response.Status.OK ).build() );
38          } 
39          
40          //log.log(Level.INFO, "Method detected: " + requestContext.getRequest().getMethod());
41          //requestContext.abortWith( Response.status( Response.Status.OK ).build() );
42      }
43  
44  }
