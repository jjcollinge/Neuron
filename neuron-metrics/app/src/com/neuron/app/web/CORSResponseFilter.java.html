1   package com.neuron.app.web;
2   
3   import java.io.IOException;
4   import java.util.logging.Level;
5   import java.util.logging.Logger;
6   
7   import javax.ws.rs.container.ContainerRequestContext;
8   import javax.ws.rs.container.ContainerResponseContext;
9   import javax.ws.rs.container.ContainerResponseFilter;
10  import javax.ws.rs.container.PreMatching;
11  import javax.ws.rs.ext.Provider;
12  
13  /**
14   * Applies a filter to any outgoing response messages. Sets the headers
15   * so that a client is aware of and can accept the data from the server.
16   * This avoids issues with browser and languages which sometimes enforce
17   * no CORS (Cross Origin Resource Sharing).
18   * @author JC
19   *
20   */
21  @Provider
22  @PreMatching
23  public class CORSResponseFilter implements ContainerResponseFilter {
24  
25      private final static Logger log = Logger.getLogger( CORSResponseFilter.class.getName() );
26      
27      public void filter(ContainerRequestContext requestContext,
28              ContainerResponseContext responseContext) throws IOException {
29          
30          log.log(Level.INFO, "Applying response filter");
31          
32          responseContext.getHeaders().add("Access-Control-Allow-Origin", "*");   
33          responseContext.getHeaders().add("Access-Control-Allow-Methods", "OPTIONS, GET, POST, PUT, DELETE");
34          responseContext.getHeaders().add("Access-Control-Allow-Credentials", "true");
35          responseContext.getHeaders().add("Access-Control-Allow-Headers", "X-Requested-With, Content-Type");
36          
37      }
38  
39  }
