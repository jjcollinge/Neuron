22  /**
23   * A representation of an in system device. Will only return
24   * a data represenation of itself including any
25   * sensors or actuators associated with it.
26   * @author JC
27   *
28   */
29  @Consumes(MediaType.APPLICATION_JSON)
30  @Produces(MediaType.APPLICATION_JSON)
31  public class DeviceResource {
32  
33      @Context
34      UriInfo uriInfo;
35      @Context
36      Request request;
37      String id;
38      private DeviceProxy proxy;
39      
40      public DeviceResource(UriInfo uriInfo, Request request, String id) {
41          
42         this.uriInfo = uriInfo;
43         this.request = request;
44         this.id = id;
45      }
46      
47      // POST: /devices/0/configure
48      @POST
49      public void configure(int refreshRate) {
50          // Grab the devices session
51          Controller controller = Controller.getApplication();
52          SessionHandler sessionHandler = (SessionHandler) controller.getActivity("SessionActivity").getService("SessionHandler");
53          Session session = sessionHandler.getSession(Integer.valueOf(id));
54          // Extract the sessions context
55          com.neuron.api.model.Context context = session.getContext();
56          proxy = new DeviceProxyFactory().getDeviceProxy(context);
57          proxy.setup(Integer.valueOf(id).intValue());
58          proxy.configureDevice(refreshRate);
59      }
60      
61      @GET
62      @Produces(MediaType.APPLICATION_JSON)
63      public Device getDevice() {
64          
65          System.out.println("Request for device");
66          DeviceDAO dao = new DeviceDAOFactory().getDeviceDAO();
67          Device device = dao.get(Integer.valueOf(id));
68          return device;
69      }
