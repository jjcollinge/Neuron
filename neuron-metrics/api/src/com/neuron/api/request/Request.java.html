1   package com.neuron.api.request;
2   
3   import java.util.ArrayList;
4   import java.util.Calendar;
5   import java.util.Date;
6   import java.util.HashMap;
7   import java.util.List;
8   import java.util.Map;
9   
10  /**
11   * Wraps a request on the system. Enables
12   * multiple protocols and formats to
13   * be supported by the system. Requests 
14   * can have an expiration period set
15   * which when reached will render the
16   * request obsolete.
17   * @author JC
18   *
19   */
20  public class Request {
21  
22      private Object data;
23      private String sourceProtocol;
24      private String sourceFormat;
25      private Calendar expiration;
26      private Map<String, List<String>> headers;
27      
28      public Request(Object data) {
29          this.data = data;
30          this.headers = new HashMap<String, List<String>>();
31          expiration = new Calendar.Builder().build();
32          expiration.setTime(new Date());
33          expiration.add(Calendar.HOUR_OF_DAY, 1);
34      }
35      
36      public Request() {
37          this.headers = new HashMap<String, List<String>>();
38          expiration = new Calendar.Builder().build();
39          expiration.setTime(new Date());
40          expiration.add(Calendar.HOUR_OF_DAY, 1);
41      }
42      
43      public void setData(Object data) {
44          this.data = data;
45      }
46      
47      public Object getData() {
48          return data;
49      }
50      
51      public void setProtocol(String protocol) {
52          this.sourceProtocol = protocol;
53      }
54      
55      public void setFormat(String format) {
56          this.sourceFormat = format;
57      }
58      
59      public String getProtocol() {
60          return this.sourceProtocol;
61      }
62      
63      public String getFormat() {
64          return this.sourceFormat;
65      }
66      
67      public void addHeader(String key, String...values) {
68          ArrayList<String> props = new ArrayList<String>();
69          for(String prop : values) {
70              props.add(prop);
71          }
72          headers.put(key, props);
73      }
74      
75      public boolean hasExpired() {
76          return expiration.after(new Date());
77      }
78      
79      public ArrayList<String> getHeader(final String key) {
80          return (ArrayList<String>) headers.get(key);
81      }
82      
83      public Map<String, List<String>>  getHeaders() {
84          return headers;
85      }
86      
87  }
