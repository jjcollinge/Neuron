1   package com.neuron.api.proxy;
2   
3   import java.util.HashMap;
4   
5   import com.neuron.api.model.Context;
6   
7   /**
8    * Masks the implementation details from the client and
9    * returns a DeviceProxy of a particular implementation
10   * based on the incoming context.
11   * @author JC
12   *
13   */
14  public class DeviceProxyFactory {
15      
16      private static HashMap<String, Class> proxies;
17      
18      public DeviceProxyFactory() {
19          if (proxies == null) {
20              proxies = new HashMap<String, Class>();
21          }
22      }
23      
24      /**
25       * Register a new proxy type with factory
26       */
27      public void registerProxy(String protocol, Class klass) {
28          proxies.put(protocol, klass);
29      }
30      
31      public DeviceProxy getDeviceProxy(Context context) {
32          
33          DeviceProxy proxy = null;   
34          String protocol = context.getProtocol();
35          Class proxyClass = proxies.get(protocol);
36          
37          if(proxyClass != null) {
38              try {
39                  proxy = (DeviceProxy) proxyClass.newInstance();
40              } catch (InstantiationException e) {
41                  e.printStackTrace();
42              } catch (IllegalAccessException e) {
43                  e.printStackTrace();
44              }
45          }
46          return proxy;
47      }
48  
49  }
