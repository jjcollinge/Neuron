3   /**
4    * Context is designed to contain all the information
5    * about a given situation. Is only concerned with
6    * the environment and situational information.
7    * Data in the context is READ-ONLY
8    * @author JC
9    *
10   */
11  public class Context {
12  
13      private String protocol;    // protocol to communicate
14      private String format;      // format of messages
15   
16      /**
17       * Holds the information about a current situation.
18       * All data is required and must be provided in the
19       * constructor.
20       * @param protocol The context's protocol
21       * @param format The context's format
22       */
23      public Context(String protocol, String format) {
24          this.protocol = protocol;
25          this.format = format;
26      }
27      
28      /**
29       * Get the protocol from the context
30       * @return String The context's protocol
31       */
32      public String getProtocol() {
33          return this.protocol;
34      }
35      
36      /**
37       * Get the format from the context
38       * @return String The context's format
39       */
40      public String getFormat() {
41          return this.format;
42      }
43      
44  }
