3   import com.neuron.api.model.Device;

5   /**
6    * A device specific implementation of an object mapper.
7    * Is used to serialize or deserialize a device from a
8    * format.
9    * @author JC
10   *
11   * @param <T> The format type from which the device it mapping to
12   */
13  public class DeviceMapper<T> implements ObjectMapper<Device, T> {
14  
15      @SuppressWarnings("rawtypes")
16      private ObjectMapperStrategy mapper;
17  
18      /**
19       * Inject a mapping strategy for a particular format type
20       */
21      @SuppressWarnings("rawtypes")
22      public void setMapperStrategy(ObjectMapperStrategy mapper) {
23          this.mapper = mapper;
24      }
25  
26      /**
27       * Serialize a device into a format type based on the mapping strategy
28       */
29      @SuppressWarnings("unchecked")
30      public T serialize(Device obj) {
31          return (T) mapper.serialize(obj);
32      }
33  
34      /**
35       * Deserialize a format type into a device based on the mapping strategy
36       */
37      @SuppressWarnings("unchecked")
38      public Device deserialize(T source) {
39          return (Device) mapper.deserialize(source);
40      }
41  }
