1   package com.neuron.api.adapters;

3   import java.util.ArrayList;
4   import java.util.HashMap;
5   import java.util.List;
6   
7   import com.neuron.api.configuration.ProtocolConfiguration;
8   
9   /**
10   * Provide the implementation details for the dapter
11   * Factory.
12   * @see com.neuron.api.adapters.AdapterFactory
13   * @author JC
14   *
15   */
16  public class AdapterFactory {
17  
18      private HashMap<String, ProtocolConfiguration> adapters;
19      private static AdapterFactory instance;
20      
21      /**
22       * Initialise the adapters collection if it has not already
23       */
24      private AdapterFactory() {
25          if (adapters == null) {
26              adapters = new HashMap<String, ProtocolConfiguration>();
27          }
28      }
29      
30      /**
31       * Singleton
32       */
33      public static AdapterFactory getFactory() {
34          if(instance == null) {
35              instance = new AdapterFactory();
36          }
37          return instance;
38      }
39  
40      /**
41       * Registers a new connector to the factory
42       */
43      public void registerAdapter(ProtocolConfiguration config) {
44          adapters.put(config.getType(), config);
45      }
46  
47      /**
48       * Create and return a new Connector of the desired protocol
49       * @return a implementation specific connector
50       */
51      public  Adapter getAdapter(String protocol) {
52  
53          ProtocolConfiguration config = adapters.get(protocol);
54  
55          if (config == null)
56              return null;
57  
58          Adapter adapter = null;
59          
60          try {
61              adapter = (Adapter) config.getMessengerClass().newInstance();
62              adapter.connect(config.getHostname(), config.getPort());
63          } catch (InstantiationException e) {
64              e.printStackTrace();
65          } catch (IllegalAccessException e) {
66              e.printStackTrace();
67          }
68  
69          return adapter;
70      }
71  
72      /**
73       * Return a catalogue of all available connector types
74       * @return a list of all connector types
75       */
76      public List<String> getCatalogue() {
77  
78          List<String> types = new ArrayList<String>();
79          for (String id : adapters.keySet()) {
80              types.add(id);
81          }
82          return types;
83      }

85  }
